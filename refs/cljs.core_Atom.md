  Object

  IAtom
  
  IEquiv
  (-equiv [o other] (identical? o other))

  IDeref

  IMeta

  IWatchable
  (-notify-watches [this oldval newval]
    (doseq [[key f] watches]
      (f key this oldval newval)))
  (-add-watch [this key f]
    (set! (.-watches this) (assoc watches key f))
    this)
  (-remove-watch [this key]

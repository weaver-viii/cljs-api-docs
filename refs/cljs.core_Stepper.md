```clj
(deftype Stepper [xform iter]
  Object
  (step [this lt]
    (loop []
      (if (and (not (nil? (.-stepper lt)))
               (.hasNext iter))
        (when-not (reduced? (xform lt (.next iter)))
          (recur))))
    (when-not (nil? (.-stepper lt))
      (xform lt))))
```

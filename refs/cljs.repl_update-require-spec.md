```
return an updated specification.
```

Source code:

```clj
(defn update-require-spec
  [specs & additions]
  (let [[before [requires & other-specs]]
        (split-with
          (fn [[x _]] (not= :require x))
          specs)
        requires'
        `(:require
           ~@(reduce
               (fn [requires spec]
                 (merge-require requires spec))
               (rest requires)
               additions))]
    (concat before [requires'] other-specs)))
```

```clj
(defn read-uuid
  [form]
  (when-not (string? form)
    (throw (RuntimeException. "UUID literal expects a string as its representation.")))
  (try
    (java.util.UUID/fromString form)
    (catch Throwable e
      (throw (RuntimeException. (.getMessage e))))))
```

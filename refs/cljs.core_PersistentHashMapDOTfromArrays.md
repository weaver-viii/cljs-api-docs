(set! cljs.core.PersistentHashMap.fromArrays
  (fn [ks vs]
    (let [len (alength ks)]
      (loop [i 0 ^not-native out (transient cljs.core.PersistentHashMap.EMPTY)]
        (if (< i len)
          (recur (inc i) (-assoc! out (aget ks i) (aget vs i)))
          (persistent! out))))))

(defn read-inst
  [form]
  (when-not (string? form)
    (throw (RuntimeException. "Instance literal expects a string for its timestamp.")))
  (try
    (inst/read-instant-date form)
    (catch Throwable e
      (throw (RuntimeException. (.getMessage e))))))
